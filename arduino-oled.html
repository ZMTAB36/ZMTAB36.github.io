    #include <U8g2lib.h>

    //构造对象

    U8G2_SSD1306_128X64_NONAME_F_SW_I2C u8g2(U8G2_R0,  4, 5, U8X8_PIN_NONE);

    void setup() {

    //初始化OLED对象

    u8g2.begin();

    //开启中文字符集

    u8g2.enableUTF8Print();

    //设置中文

    u8g2.setFont(u8g2_font_wqy16_t_gb2312);

    }

    void loop() {

      //清楚缓存区内容

      u8g2.clearBuffer();

      //位置显示

      u8g2.setCursor(10,20);

      //绘制内容

      u8g2.print("黎洪刚");

      //发送缓存区内容到OLED显示

      u8g2.sendBuffer();

    }
